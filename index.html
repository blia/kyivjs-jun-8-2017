Context (props)

			<Link href="/" isActive={isHomePage}>Home</Link>

Context (props) + css.use

			import css from 'my-css-lib';
			const activeColor =
				color => ctx => ({color: ctx.isActive ? color : 'blue'});
			css.use({ activeColor });

				a.link {
					active-color: red;
